  import pandas as pd

  def build_customer_summary(transactions):
      results = []
      for cid, grp in transactions.sort_values("customer_id").groupby(
          "customer_id", sort=False):
          running_total, tx_count = 0.0, 0
          for _, row in grp.iterrows():
              running_total += row["amount"]
              tx_count += 1
          results.append({"customer_id": cid,
              "running_total": running_total,
              "tx_count": tx_count})
      return pd.DataFrame(results)
  # explanation: RETAIN -> loop variable; FIRST./LAST. -> groupby boundaries
