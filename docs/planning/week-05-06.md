## 🎯 Goal

Build the RAPTOR semantic clustering layer — the primary research contribution of this project. Adapt RAPTOR from QA (original paper) to SAS code block grouping. This involves: (1) embedding all partitions with Nomic Embed v1.5, (2) clustering with GMM using soft-assignment threshold τ=0.72, (3) summarizing clusters with LLM (Groq/Ollama fallback), and (4) recursively building a tree until BIC converges. All nodes stored in LanceDB.

```python
import hashlib
from functools import lru_cache
from sentence_transformers import SentenceTransformer
import numpy as np

class NomicEmbedder:
    """Embed SAS code blocks using Nomic Embed v1.5 (768-dim)."""

    MODEL_NAME = "nomic-ai/nomic-embed-text-v1.5"
    DIM = 768

    def __init__(self, device: str = "cpu"):
        """
        Args:
            device: 'cpu' or 'cuda'. Auto-detect with torch.cuda.is_available().
        """
        self.model = SentenceTransformer(self.MODEL_NAME, device=device)
        self._cache: dict[str, np.ndarray] = {}

    def embed(self, text: str) -> list[float]:
        """Embed a single text. Uses SHA-256 cache to avoid re-embedding."""
        text_hash = hashlib.sha256(text.encode()).hexdigest()
        if text_hash in self._cache:
            return self._cache[text_hash].tolist()

        # Nomic Embed v1.5 expects the 'search_document:' prefix for indexing
        prefixed = f"search_document: {text}"
        embedding = self.model.encode(prefixed, normalize_embeddings=True)
        self._cache[text_hash] = embedding
        return embedding.tolist()

    def embed_batch(self, texts: list[str], batch_size: int = 32) -> list[list[float]]:
        """Embed multiple texts with batching."""
        results = []
        uncached_indices = []
        uncached_texts = []

        for i, text in enumerate(texts):
            text_hash = hashlib.sha256(text.encode()).hexdigest()
            if text_hash in self._cache:
                results.append((i, self._cache[text_hash].tolist()))
            else:
                uncached_indices.append(i)
                uncached_texts.append(f"search_document: {text}")

        if uncached_texts:
            embeddings = self.model.encode(
                uncached_texts,
                batch_size=batch_size,
                normalize_embeddings=True,
                show_progress_bar=len(uncached_texts) > 100
            )
            for idx, emb, text in zip(uncached_indices, embeddings, texts):
                text_hash = hashlib.sha256(text.encode()).hexdigest()
                self._cache[text_hash] = emb
                results.append((idx, emb.tolist()))

        # Sort by original index and return
        results.sort(key=lambda x: x[0])
        return [r[1] for r in results]

    def embed_query(self, query: str) -> list[float]:
        """Embed a query (uses 'search_query:' prefix per Nomic spec)."""
        prefixed = f"search_query: {query}"
        return self.model.encode(prefixed, normalize_embeddings=True).tolist()

    @property
    def cache_size(self) -> int:
        return len(self._cache)
```

**Key decisions**:
- `search_document:` prefix for indexing, `search_query:` for retrieval — per Nomic's task instructions
- SHA-256 cache prevents re-embedding identical code blocks (common in enterprise SAS with copy-paste macros)
- `normalize_embeddings=True` → cosine similarity = dot product (faster)
- Batch embedding for initial corpus load (50ms/block CPU, 15ms GPU)

### Task 2: GMM Clustering

**File**: `partition/raptor/clustering.py`

```python
import math
import numpy as np
from sklearn.mixture import GaussianMixture
from typing import Optional
import structlog

logger = structlog.get_logger()

class GMMClusterer:
    """
    Gaussian Mixture Model clustering for RAPTOR tree construction.
    
    Key parameters from cahier:
    - k = √N (number of components)
    - τ = 0.72 (soft-assignment threshold)
    - BIC convergence: |BIC_t - BIC_{t-1}| < ε
    """

    TAU = 0.72           # Soft-assignment threshold
    BIC_EPSILON = 0.01   # Convergence threshold
    MAX_RETRIES = 3      # Retry on ConvergenceWarning

    def cluster(
        self,
        embeddings: np.ndarray,
        max_k: Optional[int] = None,
        random_state: int = 42
    ) -> tuple[list[list[int]], float]:
        """
        Cluster embeddings using GMM.

        Args:
            embeddings: (N, 768) array of Nomic embeddings
            max_k: Override for number of clusters (default: √N)
            random_state: Random seed for reproducibility

        Returns:
            (clusters, bic): list of clusters (each = list of indices),
                             BIC score for convergence checking
        """
        n_samples = len(embeddings)
        if n_samples <= 1:
            return [[0]] if n_samples == 1 else [], 0.0

        # k = √N, minimum 2
        k = max_k or max(2, int(math.sqrt(n_samples)))
        # k cannot exceed n_samples
        k = min(k, n_samples)

        logger.info("gmm_clustering",
                     n_samples=n_samples, k=k, tau=self.TAU)

        # Fit GMM with retries
        gmm = None
        for attempt in range(self.MAX_RETRIES):
            try:
                gmm = GaussianMixture(
                    n_components=k,
                    covariance_type='full',
                    random_state=random_state + attempt,
                    max_iter=200,
                    n_init=3,
                    reg_covar=1e-5  # Prevent singular covariance
                )
                gmm.fit(embeddings)
                break
            except Exception as e:
                logger.warning("gmm_fit_retry",
                               attempt=attempt + 1, error=str(e))
                if attempt == self.MAX_RETRIES - 1:
                    logger.error("gmm_fit_failed",
                                 msg="Falling back to single cluster")
                    return [list(range(n_samples))], 0.0

        # Soft assignment: each sample may belong to multiple clusters
        # if responsibility > τ
        responsibilities = gmm.predict_proba(embeddings)  # (N, k)
        bic = gmm.bic(embeddings)

        clusters: dict[int, list[int]] = {c: [] for c in range(k)}
        for sample_idx in range(n_samples):
            assigned = False
            for cluster_idx in range(k):
                if responsibilities[sample_idx, cluster_idx] >= self.TAU:
                    clusters[cluster_idx].append(sample_idx)
                    assigned = True
            # If no cluster exceeds τ, assign to highest responsibility
            if not assigned:
                best = int(np.argmax(responsibilities[sample_idx]))
                clusters[best].append(sample_idx)

        # Remove empty clusters
        result = [members for members in clusters.values() if members]

        logger.info("gmm_clustered",
                     n_clusters=len(result),
                     bic=bic,
                     sizes=[len(c) for c in result])

        return result, bic

    def check_convergence(self, bic_prev: float, bic_curr: float) -> bool:
        """Check if BIC has converged (|BIC_t - BIC_{t-1}| < ε)."""
        return abs(bic_curr - bic_prev) < self.BIC_EPSILON
```

**Why GMM with soft assignment (not K-Means)**:
- A SAS code block implementing a DATE_ARITHMETIC merge might belong to *both* the date-handling cluster and the merge-semantics cluster
- Soft assignment (τ=0.72) allows this without duplicating the partition
- BIC (Bayesian Information Criterion) provides a principled convergence check — the paper uses it to decide when to stop recursing

### Task 3: ClusterSummarizer

**File**: `partition/raptor/summarizer.py`

```python
import hashlib
import instructor
from openai import OpenAI
from pydantic import BaseModel, Field
from typing import Optional
import structlog
import tiktoken

logger = structlog.get_logger()

class ClusterSummary(BaseModel):
    """Structured output from LLM summarization."""
    summary: str = Field(
        ...,
        description="A concise summary of what this group of SAS code blocks does together"
    )
    key_constructs: list[str] = Field(
        default_factory=list,
        description="Key SAS constructs found in this cluster (e.g., DATA step, PROC SQL, macro)"
    )
    estimated_complexity: str = Field(
        default="MODERATE",
        description="LOW | MODERATE | HIGH"
    )


class ClusterSummarizer:
    """
    Summarize a cluster of SAS code blocks using LLM.
    
    Three-tier fallback:
    1. Groq Llama 3.1 70B (best quality, rate-limited)
    2. Ollama Llama 3.1 70B local (if available)
    3. Heuristic fallback (no LLM — first/last 5 lines + keyword extraction)
    """

    # Tiktoken guard: max tokens per summarization call
    MAX_PROMPT_TOKENS = 4000
    ENCODING_NAME = "cl100k_base"  # Works for Llama tokenization approximation

    def __init__(
        self,
        groq_api_key: Optional[str] = None,
        groq_base_url: str = "https://api.groq.com/openai/v1",
        ollama_base_url: str = "http://localhost:11434/v1",
    ):
        self._summary_cache: dict[str, ClusterSummary] = {}
        self._enc = tiktoken.get_encoding(self.ENCODING_NAME)

        # Groq client (tier 1)
        self.groq_client = None
        if groq_api_key:
            self.groq_client = instructor.from_openai(
                OpenAI(api_key=groq_api_key, base_url=groq_base_url)
            )

        # Ollama client (tier 2)
        self.ollama_client = instructor.from_openai(
            OpenAI(api_key="ollama", base_url=ollama_base_url)
        )

    def summarize(self, code_blocks: list[str]) -> tuple[ClusterSummary, str]:
        """
        Summarize a cluster of code blocks.

        Returns:
            (summary, tier): summary object and tier used
                             ("groq" | "ollama_fallback" | "heuristic_fallback")
        """
        # Cache key = hash of sorted code blocks
        cache_key = hashlib.sha256(
            "||".join(sorted(code_blocks)).encode()
        ).hexdigest()

        if cache_key in self._summary_cache:
            logger.info("summary_cache_hit", cache_key=cache_key[:12])
            return self._summary_cache[cache_key], "cached"

        # Truncate code blocks to fit token limit
        truncated = self._truncate_to_token_limit(code_blocks)
        prompt = self._build_prompt(truncated)

        # Tier 1: Groq
        if self.groq_client:
            try:
                result = self.groq_client.chat.completions.create(
                    model="llama-3.1-70b-versatile",
                    messages=[{"role": "user", "content": prompt}],
                    response_model=ClusterSummary,
                    max_retries=2,
                )
                self._summary_cache[cache_key] = result
                return result, "groq"
            except Exception as e:
                logger.warning("groq_summary_failed", error=str(e))

        # Tier 2: Ollama local
        try:
            result = self.ollama_client.chat.completions.create(
                model="llama3.1:70b",
                messages=[{"role": "user", "content": prompt}],
                response_model=ClusterSummary,
                max_retries=2,
            )
            self._summary_cache[cache_key] = result
            return result, "ollama_fallback"
        except Exception as e:
            logger.warning("ollama_summary_failed", error=str(e))

        # Tier 3: Heuristic (no LLM needed)
        result = self._heuristic_summary(code_blocks)
        self._summary_cache[cache_key] = result
        return result, "heuristic_fallback"

    def _build_prompt(self, code_blocks: list[str]) -> str:
        blocks_text = "\n---\n".join(
            f"Block {i+1}:\n{block}" for i, block in enumerate(code_blocks)
        )
        return f"""You are a SAS code analysis expert. Below are {len(code_blocks)} related SAS code blocks that belong to the same semantic cluster.

Provide:
1. A concise summary (2-3 sentences) of what these blocks do *together* — their collective purpose.
2. The key SAS constructs used.
3. An estimated complexity (LOW, MODERATE, HIGH).

{blocks_text}"""

    def _truncate_to_token_limit(self, code_blocks: list[str]) -> list[str]:
        """Truncate blocks to fit within MAX_PROMPT_TOKENS."""
        total_tokens = 0
        result = []
        overhead = 200  # prompt template tokens

        for block in code_blocks:
            block_tokens = len(self._enc.encode(block))
            if total_tokens + block_tokens + overhead > self.MAX_PROMPT_TOKENS:
                # Truncate this block
                remaining = self.MAX_PROMPT_TOKENS - total_tokens - overhead
                if remaining > 50:
                    tokens = self._enc.encode(block)[:remaining]
                    result.append(self._enc.decode(tokens) + "\n/* ... truncated ... */")
                break
            result.append(block)
            total_tokens += block_tokens

        return result if result else [code_blocks[0][:500]]

    def _heuristic_summary(self, code_blocks: list[str]) -> ClusterSummary:
        """Fallback: extract keywords and structure from code blocks."""
        import re
        all_code = "\n".join(code_blocks)

        # Extract key SAS constructs
        constructs = set()
        patterns = {
            "DATA step": r'\bDATA\s+\w+',
            "PROC SQL": r'\bPROC\s+SQL\b',
            "PROC MEANS": r'\bPROC\s+MEANS\b',
            "PROC FREQ": r'\bPROC\s+FREQ\b',
            "PROC SORT": r'\bPROC\s+SORT\b',
            "PROC REG": r'\bPROC\s+REG\b',
            "PROC LOGISTIC": r'\bPROC\s+LOGISTIC\b',
            "PROC IMPORT": r'\bPROC\s+IMPORT\b',
            "Macro definition": r'%MACRO\s+\w+',
            "MERGE": r'\bMERGE\b',
            "RETAIN": r'\bRETAIN\b',
        }
        for name, pattern in patterns.items():
            if re.search(pattern, all_code, re.IGNORECASE):
                constructs.add(name)

        n_blocks = len(code_blocks)
        total_lines = sum(b.count('\n') + 1 for b in code_blocks)

        summary_text = (
            f"Cluster of {n_blocks} SAS code blocks ({total_lines} total lines). "
            f"Key constructs: {', '.join(constructs) if constructs else 'general SAS code'}."
        )

        complexity = "LOW" if total_lines < 50 else "MODERATE" if total_lines < 200 else "HIGH"

        return ClusterSummary(
            summary=summary_text,
            key_constructs=list(constructs),
            estimated_complexity=complexity
        )

    @property
    def cache_hit_rate(self) -> float:
        """Track cache effectiveness."""
        return len(self._summary_cache)  # Absolute count; compute rate in caller
```

### Task 4: RAPTORTreeBuilder — Recursive Tree Construction

**File**: `partition/raptor/tree_builder.py`

```python
import uuid
from datetime import datetime
from typing import Optional
import numpy as np
import structlog

from partition.models.raptor_node import RAPTORNode
from partition.raptor.embedder import NomicEmbedder
from partition.raptor.clustering import GMMClusterer
from partition.raptor.summarizer import ClusterSummarizer, ClusterSummary

logger = structlog.get_logger()


class RAPTORTreeBuilder:
    """
    Build a RAPTOR tree for a single SAS file.
    
    Algorithm (from Sarthi et al., adapted):
    1. Leaf nodes = PartitionIR blocks (embedded with Nomic Embed)
    2. GMM cluster at level L0 → L1 cluster nodes
    3. Summarize each cluster → embed the summary → create RAPTORNode at L1
    4. Convergence: stop if k=1 OR depth >= max_depth OR |BIC_t - BIC_{t-1}| < 0.01
    5. If not converged: recurse on L1 nodes (treat as new leaves)
    6. Final root node at top level = full-file summary
    
    Dynamic depth:
    - max_depth=3 for standard files
    - max_depth=5 when macro_density > 0.4
    """

    DEFAULT_MAX_DEPTH = 3
    MACRO_HEAVY_MAX_DEPTH = 5
    MACRO_DENSITY_THRESHOLD = 0.4

    def __init__(
        self,
        embedder: NomicEmbedder,
        clusterer: GMMClusterer,
        summarizer: ClusterSummarizer,
    ):
        self.embedder = embedder
        self.clusterer = clusterer
        self.summarizer = summarizer

    def build_tree(
        self,
        partitions: list,  # list[PartitionIR]
        file_id: str,
        macro_density: float = 0.0,
    ) -> list[RAPTORNode]:
        """
        Build the full RAPTOR tree for a file's partitions.

        Args:
            partitions: All PartitionIR objects for this file
            file_id: UUID of the source file
            macro_density: Average macro density across partitions
                           (> 0.4 → deeper tree)

        Returns:
            All RAPTORNode objects (leaf + cluster + root)
        """
        max_depth = (
            self.MACRO_HEAVY_MAX_DEPTH
            if macro_density > self.MACRO_DENSITY_THRESHOLD
            else self.DEFAULT_MAX_DEPTH
        )

        logger.info("raptor_tree_build_start",
                     file_id=file_id,
                     n_partitions=len(partitions),
                     max_depth=max_depth,
                     macro_density=macro_density)

        all_nodes: list[RAPTORNode] = []

        # Step 1: Create leaf nodes (level 0)
        leaf_nodes = self._create_leaf_nodes(partitions, file_id)
        all_nodes.extend(leaf_nodes)

        # Step 2: Recursive clustering
        current_level_nodes = leaf_nodes
        prev_bic = float('inf')
        current_depth = 0

        while current_depth < max_depth:
            current_depth += 1

            if len(current_level_nodes) <= 1:
                logger.info("raptor_single_node_stop", depth=current_depth)
                break

            # Get embeddings for current level
            embeddings = np.array([n.embedding for n in current_level_nodes])

            # Cluster
            clusters, bic = self.clusterer.cluster(embeddings)

            # Check convergence
            if self.clusterer.check_convergence(prev_bic, bic):
                logger.info("raptor_bic_converged",
                           depth=current_depth,
                           bic_prev=prev_bic,
                           bic_curr=bic)
                break
            prev_bic = bic

            # Create cluster nodes at this level
            cluster_nodes = []
            for cluster_idx, member_indices in enumerate(clusters):
                members = [current_level_nodes[i] for i in member_indices]
                cluster_node = self._create_cluster_node(
                    members=members,
                    level=current_depth,
                    cluster_label=cluster_idx,
                    file_id=file_id,
                )
                cluster_nodes.append(cluster_node)
                all_nodes.append(cluster_node)

            logger.info("raptor_level_complete",
                        depth=current_depth,
                        n_clusters=len(cluster_nodes))

            current_level_nodes = cluster_nodes

        # Create root node (if we have multiple nodes at the top)
        if len(current_level_nodes) > 1:
            root = self._create_root_node(current_level_nodes, file_id, current_depth + 1)
            all_nodes.append(root)
        elif len(current_level_nodes) == 1:
            # Promote the single remaining node to root
            current_level_nodes[0].level = current_depth

        # Back-link partitions → RAPTOR nodes
        self._backlink_partitions(partitions, all_nodes)

        logger.info("raptor_tree_complete",
                     file_id=file_id,
                     total_nodes=len(all_nodes),
                     tree_depth=current_depth)

        return all_nodes

    def _create_leaf_nodes(
        self,
        partitions: list,
        file_id: str,
    ) -> list[RAPTORNode]:
        """Create level-0 leaf nodes from partitions."""
        # Batch embed all raw_code
        codes = [p.raw_code for p in partitions]
        embeddings = self.embedder.embed_batch(codes)

        leaf_nodes = []
        for partition, embedding in zip(partitions, embeddings):
            node = RAPTORNode(
                node_id=uuid.uuid4(),
                level=0,
                summary=f"Leaf: {partition.partition_type.value} "
                        f"(lines {partition.line_start}-{partition.line_end})",
                summary_tier="skipped",  # Leaf nodes don't get LLM summaries
                embedding=embedding,
                child_ids=[],
                cluster_label=None,
                file_id=uuid.UUID(file_id) if isinstance(file_id, str) else file_id,
                partition_ids=[str(partition.partition_id)],
            )
            leaf_nodes.append(node)

        return leaf_nodes

    def _create_cluster_node(
        self,
        members: list[RAPTORNode],
        level: int,
        cluster_label: int,
        file_id: str,
    ) -> RAPTORNode:
        """Create a cluster node by summarizing and embedding its members."""
        # Collect all raw partition codes from member subtrees
        code_blocks = []
        all_partition_ids = []
        for member in members:
            all_partition_ids.extend(member.partition_ids)
            # Use the member summary as input to the higher-level summary
            code_blocks.append(member.summary)

        # Summarize
        summary_obj, tier = self.summarizer.summarize(code_blocks)

        # Embed the summary text
        summary_embedding = self.embedder.embed(summary_obj.summary)

        return RAPTORNode(
            node_id=uuid.uuid4(),
            level=level,
            summary=summary_obj.summary,
            summary_tier=tier,
            embedding=summary_embedding,
            child_ids=[str(m.node_id) for m in members],
            cluster_label=cluster_label,
            file_id=uuid.UUID(file_id) if isinstance(file_id, str) else file_id,
            partition_ids=all_partition_ids,
        )

    def _create_root_node(
        self,
        top_nodes: list[RAPTORNode],
        file_id: str,
        level: int,
    ) -> RAPTORNode:
        """Create the single root node summarizing the entire file."""
        summaries = [n.summary for n in top_nodes]
        all_partition_ids = []
        for n in top_nodes:
            all_partition_ids.extend(n.partition_ids)

        summary_obj, tier = self.summarizer.summarize(summaries)
        summary_embedding = self.embedder.embed(summary_obj.summary)

        return RAPTORNode(
            node_id=uuid.uuid4(),
            level=level,
            summary=summary_obj.summary,
            summary_tier=tier,
            embedding=summary_embedding,
            child_ids=[str(n.node_id) for n in top_nodes],
            cluster_label=None,
            file_id=uuid.UUID(file_id) if isinstance(file_id, str) else file_id,
            partition_ids=all_partition_ids,
        )

    def _backlink_partitions(
        self,
        partitions: list,
        all_nodes: list[RAPTORNode],
    ):
        """Set raptor_leaf_id, raptor_cluster_id, raptor_root_id on partitions."""
        # Build lookup: partition_id → leaf node
        leaf_lookup = {}
        for node in all_nodes:
            if node.level == 0 and len(node.partition_ids) == 1:
                leaf_lookup[node.partition_ids[0]] = node

        # Find root (highest level)
        root_node = max(all_nodes, key=lambda n: n.level) if all_nodes else None

        # Find L1 cluster for each partition
        l1_clusters = [n for n in all_nodes if n.level == 1]

        for partition in partitions:
            pid = str(partition.partition_id)

            # Leaf
            if pid in leaf_lookup:
                partition.raptor_leaf_id = str(leaf_lookup[pid].node_id)

            # L1 cluster
            for cluster in l1_clusters:
                if pid in cluster.partition_ids:
                    partition.raptor_cluster_id = str(cluster.node_id)
                    break

            # Root
            if root_node:
                partition.raptor_root_id = str(root_node.node_id)
```

```python
import os
from partition.base_agent import BaseAgent
from partition.raptor.embedder import NomicEmbedder
from partition.raptor.clustering import GMMClusterer
from partition.raptor.summarizer import ClusterSummarizer
from partition.raptor.tree_builder import RAPTORTreeBuilder

class RAPTORPartitionAgent(BaseAgent):
    """
    Agent #7: RAPTOR semantic clustering.
    
    Orchestrates embedding → clustering → summarization → tree building
    for each file's partition set.
    """
    agent_name = "RAPTORPartitionAgent"

    def __init__(self, trace_id=None, device="cpu"):
        super().__init__(trace_id)

        self.embedder = NomicEmbedder(device=device)
        self.clusterer = GMMClusterer()
        self.summarizer = ClusterSummarizer(
            groq_api_key=os.getenv("GROQ_API_KEY"),
        )
        self.tree_builder = RAPTORTreeBuilder(
            embedder=self.embedder,
            clusterer=self.clusterer,
            summarizer=self.summarizer,
        )

    async def process(self, partitions: list, file_id: str) -> list:
        """
        Build RAPTOR tree for a file's partitions.
        
        Args:
            partitions: All PartitionIR objects for one file
            file_id: Source file UUID
            
        Returns:
            list[RAPTORNode]: All tree nodes (leaf + cluster + root)
        """
        if not partitions:
            self.logger.warning("raptor_no_partitions", file_id=file_id)
            return []

        # Compute average macro density for dynamic depth
        macro_density = sum(
            1 for p in partitions if p.has_macros
        ) / len(partitions)

        self.logger.info("raptor_agent_start",
                         file_id=file_id,
                         n_partitions=len(partitions),
                         macro_density=macro_density)

        try:
            nodes = self.tree_builder.build_tree(
                partitions=partitions,
                file_id=file_id,
                macro_density=macro_density,
            )

            self.logger.info("raptor_agent_complete",
                             file_id=file_id,
                             total_nodes=len(nodes),
                             leaf_count=sum(1 for n in nodes if n.level == 0),
                             cluster_count=sum(1 for n in nodes if n.level > 0))

            return nodes

        except Exception as e:
            self.logger.error("raptor_agent_failed",
                              file_id=file_id,
                              error=str(e))
            # Fallback: return leaf-only nodes (flat partition)
            return self.tree_builder._create_leaf_nodes(partitions, file_id)
```

**File**: `partition/raptor/lancedb_writer.py`

```python
import lancedb
import pyarrow as pa
from datetime import datetime
from typing import Optional

# Schema for raptor_nodes table
RAPTOR_SCHEMA = pa.schema([
    pa.field('node_id',        pa.string()),
    pa.field('level',          pa.int32()),
    pa.field('summary',        pa.string()),
    pa.field('summary_tier',   pa.string()),
    pa.field('embedding',      pa.list_(pa.float32(), 768)),
    pa.field('child_ids',      pa.string()),     # JSON array
    pa.field('cluster_label',  pa.int32()),
    pa.field('file_id',        pa.string()),
    pa.field('partition_ids',  pa.string()),     # JSON array
    pa.field('created_at',     pa.string()),
])


class RAPTORLanceDBWriter:
    """Write RAPTOR nodes to LanceDB with cosine IVF index."""

    TABLE_NAME = "raptor_nodes"
    NUM_PARTITIONS = 32  # IVF index parameter

    def __init__(self, db_path: str = "lancedb_data"):
        self.db = lancedb.connect(db_path)

    def upsert_nodes(self, nodes: list) -> int:
        """
        Write RAPTOR nodes to LanceDB.
        Creates the table on first call, appends on subsequent calls.
        
        Returns:
            Number of nodes written
        """
        import json

        records = []
        for node in nodes:
            records.append({
                "node_id": str(node.node_id),
                "level": node.level,
                "summary": node.summary,
                "summary_tier": node.summary_tier,
                "embedding": node.embedding,
                "child_ids": json.dumps([str(c) for c in node.child_ids]),
                "cluster_label": node.cluster_label if node.cluster_label is not None else -1,
                "file_id": str(node.file_id),
                "partition_ids": json.dumps(node.partition_ids),
                "created_at": datetime.utcnow().isoformat(),
            })

        if self.TABLE_NAME in self.db.table_names():
            table = self.db.open_table(self.TABLE_NAME)
            table.add(records)
        else:
            table = self.db.create_table(
                self.TABLE_NAME,
                data=records,
                schema=RAPTOR_SCHEMA,
            )

        # Create/update IVF index (only if enough data)
        try:
            if len(table) >= self.NUM_PARTITIONS * 2:
                table.create_index(
                    metric="cosine",
                    num_partitions=self.NUM_PARTITIONS,
                    num_sub_vectors=16,
                    replace=True,
                )
        except Exception:
            pass  # Index creation fails on small tables — that's OK

        return len(records)

    def query_similar(
        self,
        query_embedding: list[float],
        k: int = 5,
        level: Optional[int] = None,
        file_id: Optional[str] = None,
    ) -> list[dict]:
        """
        Query the RAPTOR tree for similar nodes.
        
        Used by TranslationAgent in L3 for context retrieval.
        """
        if self.TABLE_NAME not in self.db.table_names():
            return []

        table = self.db.open_table(self.TABLE_NAME)
        query = table.search(query_embedding).limit(k)

        # Optional filters
        if level is not None:
            query = query.where(f"level = {level}")
        if file_id:
            query = query.where(f"file_id = '{file_id}'")

        results = query.to_list()
        return results

    def count_nodes(self, file_id: Optional[str] = None) -> int:
        """Count nodes in the RAPTOR table."""
        if self.TABLE_NAME not in self.db.table_names():
            return 0
        table = self.db.open_table(self.TABLE_NAME)
        if file_id:
            return len(table.search().where(f"file_id = '{file_id}'").to_list())
        return len(table)
```

## Common Pitfalls

| Pitfall | How to Avoid |
|---------|-------------|
| Nomic Embed requires prefixes (`search_document:` / `search_query:`) | Always use the correct prefix — missing it degrades retrieval quality by ~15% |
| GMM ConvergenceWarning on small clusters | `reg_covar=1e-5` + retry with shifted seed |
| CUDA OOM during batch embedding | Set `PYTORCH_CUDA_ALLOC_CONF=max_split_size_mb:128`, reduce batch_size |
| Groq rate limit (30 req/min) | The summarizer uses `max_retries=2`; exponential backoff happens in `instructor` |
| Tiktoken approximate for Llama | `cl100k_base` overestimates Llama tokens by ~5% — safe side |
| LanceDB IVF index needs ≥ 64 vectors | `create_index` silently skipped when table is too small |
| Summary cache key must sort blocks | Unsorted blocks = different cache key for same cluster |
